    <script>
        // Mock data and state management
        let currentServingNumber = 1;
        let nextQueueNumber = 2;
        let servedToday = 0;
        let isLoggedIn = false;
        let recentCustomers = [];

        // Page navigation
        function showPage(pageId) {
            document.querySelectorAll('.page').forEach(page => page.classList.remove('active'));
            document.getElementById(pageId).classList.add('active');
        }

        // Mobile menu functions
        function toggleMobileMenu() {
            const hamburger = document.querySelector('.hamburger');
            const mobileMenu = document.getElementById('mobileMenu');
            
            hamburger.classList.toggle('active');
            mobileMenu.classList.toggle('active');
        }

        function closeMobileMenu() {
            const hamburger = document.querySelector('.hamburger');
            const mobileMenu = document.getElementById('mobileMenu');
            
            hamburger.classList.remove('active');
            mobileMenu.classList.remove('active');
        }

        // Close mobile menu when clicking outside
        document.addEventListener('click', function(event) {
            const hamburger = document.querySelector('.hamburger');
            const mobileMenu = document.getElementById('mobileMenu');
            const header = document.querySelector('header');
            
            if (!header.contains(event.target) && mobileMenu.classList.contains('active')) {
                closeMobileMenu();
            }
        });

        // Ultra-simple queue management
        document.getElementById('queueForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const name = document.getElementById('customerName').value;
            const service = document.getElementById('serviceType').value;
            const phone = document.getElementById('phone').value;
            
            // Give customer their number
            const customerNumber = nextQueueNumber;
            nextQueueNumber++;
            
            // Store customer info
            const customer = {
                number: customerNumber,
                name: name,
                service: service,
                phone: phone,
                timestamp: new Date().toLocaleTimeString()
            };
            
            recentCustomers.unshift(customer);
            if (recentCustomers.length > 10) {
                recentCustomers.pop(); // Keep only last 10
            }
            
            // Update display
            document.getElementById('yourNumber').textContent = customerNumber;
            document.getElementById('queueMessage').textContent = `Your number is #${customerNumber}. Currently serving #${currentServingNumber}`;
            
            // Clear form
            document.getElementById('queueForm').reset();
            
            // Update admin dashboard if open
            updateAdminDisplay();
            
            alert(`Your queue number is #${customerNumber}\n\nCurrently serving: #${currentServingNumber}\nCustomers ahead: ${customerNumber - currentServingNumber}`);
        });

        // Update all displays
        function updateAdminDisplay() {
            // Update admin stats
            document.getElementById('nowServingAdmin').textContent = currentServingNumber;
            document.getElementById('nextNumber').textContent = nextQueueNumber;
            document.getElementById('todayServed').textContent = servedToday;
            document.getElementById('currentServingDisplay').textContent = currentServingNumber;
            
            // Update customer display
            document.getElementById('nowServing').textContent = currentServingNumber;
            
            // Update recent customers list
            const recentDiv = document.getElementById('recentCustomers');
            if (recentCustomers.length === 0) {
                recentDiv.innerHTML = '<p style="text-align: center; color: #666;">No customers yet today</p>';
            } else {
                recentDiv.innerHTML = recentCustomers.map(customer => `
                    <div style="padding: 0.5rem 0; border-bottom: 1px solid #eee;">
                        <strong>#${customer.number}</strong> - ${customer.name} (${customer.service})
                        <br><small>${customer.timestamp}</small>
                    </div>
                `).join('');
            }
        }

        // Admin functions
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const username = document.getElementById('username').value;
            const password = document.getElementById('password').value;
            
            if (username === 'admin' && password === 'password') {
                isLoggedIn = true;
                showPage('dashboard');
                updateAdminDisplay();
            } else {
                alert('Invalid credentials. Try: admin / password');
            }
        });

        function logout() {
            isLoggedIn = false;
            showPage('home');
        }

        function callNext() {
            currentServingNumber++;
            servedToday++;
            updateAdminDisplay();
            
            // Show confirmation
            alert(`Now serving customer #${currentServingNumber}`);
        }

        function resetQueue() {
            if (confirm('Reset queue to #1? This will clear all progress.')) {
                currentServingNumber = 1;
                nextQueueNumber = 2;
                servedToday = 0;
                recentCustomers = [];
                updateAdminDisplay();
            }
        }

        function goBack() {
            if (currentServingNumber > 1) {
                currentServingNumber--;
                servedToday = Math.max(0, servedToday - 1);
                updateAdminDisplay();
            }
        }

        // Initialize
        updateAdminDisplay();
    </script>
